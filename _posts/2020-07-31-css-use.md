/* style.css */
html, body, div, h1, h2, h3, h4, h5, h6, p, span, img, input {

	padding: 0;
}

	font-size: 19px;

	width: 800px;

	margin-left: -400px;

	overflow: hidden;
	z-index: 2;
	padding: 50px;
	box-sizing: border-box;
}

.content::before {
	content: '';
	position: absolute;
	top: 0;
	left: 0;
	right: 0;
	bottom: 0;

	text-align: center;
